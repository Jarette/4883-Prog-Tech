class Solution {
public:
    double myPow(double x, int n) {
        if (x < 0) {
            x = x * -1;
            if (n % 2 != 0) {
                long double Ln = log(x);
                long double new_exp = Ln * n;
                long double result = exp(new_exp);
                return result * -1;
            }
        }
        long double Ln = log(x);
        long double new_exp = Ln * n;
        long double result = exp(new_exp);
        return result;
    }
};
