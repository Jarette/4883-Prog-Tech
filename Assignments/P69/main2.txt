class Solution {
public:
    int mySqrt(int x) {
        int odds = 1;
        int count = 0;
        if (x < 0) {
            x = x * -1;
        }
        while (x > 0) {
            x = x - odds;
            if (x < 0) {
                break;
            }
            odds = odds + 2;
            count++;
        }
        cout << count;
        return count;
    }
};
